\begin{exercise}
\topic{naturais!Peano}
\topic{recursão}
Defina recursivamente a função $d : \nats \to \nats$ que dobra sua entrada. Verifique que $d(3) = 6$. 
\end{exercise}

\begin{exercise}
\topic{naturais!Peano}
\topic{recursão}
Defina recursivamente a multiplicação $\_\cdot\_: \nats^2 \to \nats$. Verifique que $2 \cdot 3 = 6$.
\end{exercise}
