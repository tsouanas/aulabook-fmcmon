\begin{exercise}
	Recorde a definição para $a^n$:
	\begin{align*}
		a^0 &= e \\
		a^{n+1} &= a \cdot a^n
	\end{align*}

	Abaixo temos uma definição alternativa, executando o passo recursivo à esquerda:
	\begin{align*}
		a^0 &= e \\
		a^{n+1} &= a^n \cdot a
	\end{align*} 

	Mostre que essa nova definição é equivalente à usual. 
\end{exercise}
