\begin{problem}
\label{nicebij}
\topic{functions!bijections}
\topic{functions!preimage}
Seja $f: A \bijto B$.
\end{problem}

\begin{definition}
Seja $n\in\nats$.  O $n$ é \dterm{lindo}
sse ele não é feio.
\end{definition}

\begin{exercise}
\topic{induction!strong}
Usando o \ref{nicebij} (p.~\pageref{nicebij}), prove that for all $n\in\nats$, $n$ is \emph{cool}.
\end{exercise}

\begin{homework}
Do 30 sit-ups and 20 push-ups.
\end{homework}
