\begin{problem}
\label{nicebij}
\topic{functions!bijections}
\topic{functions!preimage}
Seja $f: A \bijto B$.
\end{problem}

\begin{definition}
Seja $n\in\nats$.  O $n$ é \dterm{lindo}
sse ele não é feio.
\end{definition}

\begin{exercise}
\topic{induction!strong}
Usando o \ref{nicebij} (p.~\pageref{nicebij}), prove that for all $n\in\nats$, $n$ is \emph{cool}.
\end{exercise}

\begin{homework}
Do 30 sit-ups and 20 push-ups.
\end{homework}

\begin{notes}
    Veja no~\cite[\S2.3]{velleman} e no~\cite[cap.~4]{pinter}
    para mais informações sobre esse tipo de coisa.
\end{notes}

\begin{exercise}
\topic{induction}
Demonstre que a adição definida no exercício anterior é associativa.
Isso é, mostre que para todo $m, n, p$ naturais, $m + (n + p) = (m + n) + p$.
\end{exercise}

\begin{definition}[Lema 1 da adição]
	\label{lema1}
	Para todo $n$ natural, $n + 0 = n$.
\end{definition}

\begin{definition}[Lema 2 da adição]
	\label{lema2}
	Para todo $m, n$ naturais, $S(m + n) = m + Sn$.
\end{definition}

\begin{homework}
\topic{induction}

\end{homework}
