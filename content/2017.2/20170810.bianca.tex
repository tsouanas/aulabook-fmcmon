\begin{exercise}
\topic{indução}
	A sequência Fibonacci é a sequência de inteiros $F_1, F_2, F_3,\ldots$ definida como segue
	$$
	\left\{
		\begin{aligned}
			&F_1 = 1;\\
			&F_2 = 1;\\
			&F_{n + 2} = F_{n + 1} + F_n
		\end{aligned}
	\right.
	$$
	para todos os inteiros positivos $n$. Use indução para provar que para todo $n > 0$,
	$$
	F_{n + 1} F_{n + 2} - F_n F_{n + 3} = (-1)^n
	$$
\end{exercise}
