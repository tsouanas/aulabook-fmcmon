\begin{exercise}
\topic{funções!bijetividade}
Mostre que, se $f : A \to B$ e $g : B \to C$ possuem inversa, então $g \compose f : A \to C$ também possui inversa.
\end{exercise}

\begin{exercise}
\topic{funções!bijetividade}
Sejam $f : A \to B$ e $g : B \to A$ tais que $g \compose f = id_A$.
Prove que:
	\begin{enumerate}
		\item[a)] $g$ é sobrejetora; 
		\item[b)] $f$ é injetora.
	\end{enumerate}
\end{exercise}
