\begin{exercise}
  (Lema de Euclides) Sejam $a,b \in \ints$ e $p$ primo. Se $p \divs ab$, ent\~ao
  $p \divs a$ ou $p \divs b$.
\end{exercise}

\begin{exercise}
  Sejam $a \in \ints$ e $p$ primo. Se $p \not\divs a$, ent\~ao $gcd{a,p} = 1$.
\end{exercise}

\begin{homework}
  Sejam $a \in \ints$ e $p$ primo. Se $a \divs p$, ent\~ao $a=1$ ou $a=p$.
\end{homework}
