\begin{exercise}
    Sejam $a, b, p \in \ints$. Se $p$ é primo e $p \divs ab$, então $p \divs a$
    ou $p \divs b$.
\end{exercise}

\begin{exercise}[Teorema Fundamental da Aritmética]
    Todo $n \in \nats$, com $n > 1$, pode ser escrito como um único produtório
    de primos, a menos de ordem.
\end{exercise}
