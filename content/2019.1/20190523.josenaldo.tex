\begin{exercise}
\topic{teoria de números!divisibilidade}
    Para $n \ge 2$ e $k$ um inteiro positivo qualquer temos que $(n-1) \div n^k - 1$.
\end{exercise}

\topic{teoria de números!divisibilidade}

\end{exercise}
