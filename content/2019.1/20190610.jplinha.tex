\begin{exercise}
    Mostrar que $2(p-3)! \cong_p -1$ para $p$ primo ímpar.
\end{exercise}

\begin{exercise}
    Se $a$ e $b$ são inteiros, temos que $a \cong_m b$ se, e somente se, $a = b
    + km$ para algum $k$ inteiro.
\end{exercise}

\begin{homework}
    Mostre que para todo $a \in \ints$, $a^7 \cong_{27} a$.
\end{homework}
