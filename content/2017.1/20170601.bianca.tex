\begin{exercise}
\topic{teoria de números}
	Se $a < 0$ e $a\divides b$, então $\gcd a b = a$.
\end{exercise}

\begin{exercise}
\topic{teoria de números}
	Se $ab\cong 0\pmod p$, onde $p$ é um primo, então $a\cong 0\pmod p$ ou $b\cong 0\pmod p$.
\end{exercise}

\begin{exercise}
\topic{teoria de números}
	$ax + by = c$ tem soluções sse $\gcd a b\divides c$.
\end{exercise}
