\begin{definition}
    Seja $x\in\reals$. O $x$ é \dterm{irracional} sse existem $p,q \in\ints$ tais que:
    $$
        x = \frac p q.
    $$
\end{definition}

\begin{exercise}
\label{sqrt2}%
\topic{irracionalidade}%
    Prove que $\sqrt 2$ é irracional.
\end{exercise}

\begin{exercise}
    Que alterações devem ser feitas no \ref{sqrt2} para provar que $\sqrt 3$ é irracional?
\end{exercise}

\begin{exercise}
    Mostre que existem irracionais $x,y$ com $x^y$ racional.
\end{exercise}
