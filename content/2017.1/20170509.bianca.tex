\begin{exercise}
\topic{indução}%
\topic{recursão}%
\topic{recorrência}%
    Uma sequência $a_0, a_1, a_2, \ldots$ é definida recursivamente da seguinte forma:
    %\begin{center}
    %    $a_0 = 0$;
    %    para todo $n\in\nats, a_{n + 1} = 2a_n + n$.
    %\end{center}
    %Como eu faço para as duas linhas ficarem alinhadas no centro?
    %assim:
    \begin{align*}
        a_0 &= 0 \\
        \text{para todo $n\in\nats$,}\quad
        a_{n+1} &= 2a_n + n
    \end{align*}
    Prove que para todo $n\in\nats$, $a_n = 2^n - n - 1$.
\end{exercise}

\begin{exercise}
\topic{indução!forte}
    Para todos os naturais $n$ e $m$, se $m > 0$, então existem naturais $q$ e $r$ tais que $n = mq + r$ e $r < m$.
\end{exercise}

\begin{exercise}
\topic{indução!forte}
    Todo inteiro $n > 1$ é primo ou um produto de primos.
\end{exercise}
