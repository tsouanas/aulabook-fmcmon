\begin{exercise}
\label{sum_up_to_n}
\topic{indução}%
Prove que para todo $n \in\nats$
$$
0 + 1 + 2 + \dotsb + n = \frac {n(n+1)} 2.
$$
\end{exercise}

\begin{exercise}
\topic{indução}%
Encontre uma fórmula para $1 + 3 + 5 + \dotsb + (2n - 1)$ para $n \leq 1$ e prove que sua fórmula é correta.
\end{exercise}

\begin{exercise}
\topic{indução}%
Prove que para todo $n \in\ints$, tal que $n \geq 5$
$$
n^2 < 2^n.
$$
\end{exercise}
