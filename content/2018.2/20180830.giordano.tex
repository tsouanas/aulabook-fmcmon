\begin{exercise}
\topic{conjuntos}%
Seja $A$ conjunto, prove que $\emptyset \subseteq A$
\end{exercise}

\begin{exercise}
\topic{conjuntos}%
Seja $A$ conjunto, prove que $A \subseteq A$
\end{exercise}

\begin{exercise}
\topic{conjuntos}%
Sejam $A$ e $B$ conjuntos, prove que $A \subseteq B$ se e somente se $A \cap B = A$
\end{exercise}
