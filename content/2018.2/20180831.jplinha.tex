\begin{exercise}
    \topic{conjuntos}%
    Sejam $\{A_n\}_n$ e $\{B_n\}_n$ duas sequências de conjuntos t.q.
    $$ \text{para todo número par } m, A_m \subseteq B_{m/2} $$
\end{exercise}

    Prove que 
    $$
    C \setminus \Inter_{n = 0}^\infty A_n = 
    \Inter_{n = 0}^\infty (C \setminus A_n)
    $$
\end{exercise}
