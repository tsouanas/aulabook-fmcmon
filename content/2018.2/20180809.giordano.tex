\begin{exercise}
\topic{teoria de números!divisibilidade}%
	Demonstre que para quaisquer inteiros $a$ e $b$, valem
	\begin{enumerate}[(i)]
	\item $0 \divides a$
	\item $a \divides a$
	\end{enumerate}
\end{exercise}

\begin{exercise}
\topic{teoria de números}%
	Demonstre que para qualquer inteiro $n$, vale que $n$ é par se e somente se $n^2$ é par.
\end{exercise}
