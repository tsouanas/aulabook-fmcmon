\begin{exercise}
\topic{funções!injetividade}
\topic{funções!ponto fixo}
\topic{funções!pré-imagem}
Sejam $f : A \to A$ e $F = \{x \in A : f(x) = x\}$. Mostre que:

\begin{enumerate}[a)]
\item $f[F] \subseteq F$;
\item $f^{-1}[F] \subseteq F$, se $f$ é injetiva.
\end{enumerate}
\end{exercise}
