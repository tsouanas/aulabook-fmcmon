\begin{exercise}
\topic{conjuntos!cardinalidade}
    Mostre que $<_c$ é
    \begin{enumerate}[(i)]
        \item irreflexiva
        \item transitiva
    \end{enumerate}
\end{exercise}

\begin{exercise}
\topic{conjuntos!cardinalidade}
    Mostre que $A \leq_c B \iff (\exists f)[f : A \injto B]$.
\end{exercise}

\begin{exercise}
\topic{conjuntos!cardinalidade}
    Mostre que se $A =_c B$, então $\powerset A =_c \powerset B$.
\end{exercise}
