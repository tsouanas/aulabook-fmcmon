\begin{exercise}
\topic{conjuntos!cardinalidade}
    Mostre que $A$ é contável sse $A \leq_c \nats$.    
\end{exercise}

\begin{exercise}
\topic{conjuntos!cardinalidade}
    Mostre que $A$ é contável sse $A = \emptyset$ ou $A$ tem uma enumeração, isto é, uma sobrejeção $g : \nats \surjto A$.    
\end{exercise}
