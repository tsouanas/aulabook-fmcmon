\begin{exercise}
\topic{conjuntos}
Sejam $P$, $Q$ e $R$ conjuntos. Considere as seguintes asserções:
\begin{itemize}
  \item[(a)] $P \subseteq R$
  \item[(b)] $P \union (Q \inter R) = (P \union Q) \inter R$
  \item[(c)] $(P \inter Q) \union R = R$
\end{itemize}

Verifique (isto é, demonstre ou refute) as seguintes asserções:
\begin{itemize}
  \item (1a) implica (1b)
  \item (1b) implica (1a)
  \item (1a) implica (1c)
  \item (1c) implica (1a)
\end{itemize}
\end{exercise}
