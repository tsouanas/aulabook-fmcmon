\begin{exercise}
\topic{funções}
Prove que:
\begin{itemize}
    \item $f : B \to C$ é injetora sse para todos $h,k : A -> B$, se $f \com h = f \com k$ então $h = k$.
    \item $g : A \to B$ é sobrejetora sse para todos $h,k : B -> C$, se $h \com g = k \com g$ entaõ $h = k$.
\end{itemize}
\end{exercise}
